# 布局

# 横向布局

body:Row(//关键字
  children:<Widget>[
    crossAxisAlignment:CrossAxisAlignment.end,//对齐方式, 上中下
    Container(
      width:200,//宽度超过会报提示
      height:200,
      color:Color.red,
    ),
    Container(
      width:400,
      height:300,
      color:Color.yellow,
    )
  ]
)

# 垂直对齐

body:Column(//关键字
  children:<Widget>[
    crossAxisAlignment:CrossAxisAlignment.start,//对齐方式,左中右
    Container(
      width:200,//宽度超过会报提示
      height:200,
      color:Color.red,
    ),
    Container(
      width:400,
      height:300,
      color:Color.yellow,
    )
  ]
)

# 弹性布局

body:Flex(//关键字
  direction: Axis.horizontal,//必须的，对齐方式横向
  direction: Axis.vertical,//必须的，对齐方式竖向，默认全屏高度
  children: <Widget>[
     Container(//固定宽度，其它剩余宽度弹性
      width:400,
      height:300,
      color:Color.yellow,
    ),
    Expanded(
      flex:1,//占比
      child: Container(
        height:300,
        color:Color.yellow,
      )
    ),
    Expanded(
      flex:2,
      child: Container(
        height:300,
        color:Color.yellow,
      )
    ),
    Expanded(
      flex:3,
      child: Container(
        height:300,
        color:Color.yellow,
      )
    ),
  ]
)

# 定位

body:SizedBox(
  height:400,
  width:400,
  child:Stack(
    alignment:Alignment.center,//排列方式
    children:<Widget>[
      Positioned(
        left:10,//定位权重比宽高高
        top:10,
        child:Container(
          height:100,
          width:100,
          color:Color.red,
        )
      )
    ]
  )
)

# 流式布局

body:Wrap(//关键字
  spacing:25,//间隙
  runSpacing:10,//竖值间隙
  alignment:WrapAlignment.end,//换行后的对齐方式, 左中右
  children:<Widget>[
    Container(
      width:200,//宽度超过会报提示
      height:200,
      color:Color.red,
    ),
    Container(
      width:400,
      height:300,
      color:Color.yellow,
    )
  ]
)
